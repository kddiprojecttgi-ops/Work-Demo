Database Schema Installation Steps

Follow these simple steps to live-connect your database queries:

  1. Log in to your Supabase project dashboard: https://supabase.com/dashboard ↗
  2. Select the project matching your provided link.
  3. Go to the SQL Editor tab on the left-side navigation rail.
  4. Click New Query.
  5. Copy the system SQL migration script below, paste it, and click Run (or press cmd+Enter).
  6. Refresh this application! Your data is fully linked up.
Setup SQL Query Script 
-- Create truck_options table for dynamic dropdowns
CREATE TABLE IF NOT EXISTS truck_options (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  type TEXT NOT NULL, -- 'truckNumbers', 'owners', 'warehouses'
  value TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  CONSTRAINT unique_type_value UNIQUE (type, value)
);

ON CONFLICT (type, value) DO NOTHING;

-- Create truck_records table for deliveries
CREATE TABLE IF NOT EXISTS truck_records (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  date DATE NOT NULL,
  truck_number TEXT NOT NULL,
  owner TEXT NOT NULL,
  warehouse_number TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
